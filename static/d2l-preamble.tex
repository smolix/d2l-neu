% LaTeX preamble for "Dive into Deep Learning" PDF builds.
% Referenced by tools/gen_pdf.py via `include-in-header`.
% Quarto already loads fontspec/hyperref/amsmath/amssymb/graphicx/longtable/booktabs,
% so this file only adds project-specific tweaks.

% Allow line breaks inside long verbatim spans (URLs, code lines).
\usepackage{fvextra}
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{commandchars=\\\{\},breaklines,breakanywhere,fontsize=\small}
% Highlighting (above) wraps source code, but pandoc emits cell *output* as the
% kernel's plain `verbatim`, which clips at the right margin. Redefine it as a
% breakable fvextra Verbatim so wide output — generated-text samples, dir() API
% dumps, profiler tables — wraps instead of running off the page. (Pandoc always
% puts \end{verbatim} on its own line, which fvextra requires.)
\DefineVerbatimEnvironment{verbatim}{Verbatim}{breaklines,breakanywhere,fontsize=\small}

% Better float placement for figures inside narrow sections.
\usepackage{float}

% mathtools (extends amsmath) provides the small parenthesized-matrix
% environments — pandoc's LaTeX writer normalizes
% \left(\begin{smallmatrix}…\end{smallmatrix}\right) into \begin{psmallmatrix},
% which is undefined without this. Load after amsmath (which Quarto loads).
\usepackage{mathtools}

% Allow underscores in identifiers without escaping.
\usepackage[strings]{underscore}

% Show subsections in the TOC.
\setcounter{tocdepth}{1}

% Looser hyphenation to reduce overfull \hbox warnings in code-heavy paragraphs.
\sloppy
